Below is a code to do different math operations with just one call.
Call: 
_math_(2,3,"+") // calls this function

function _math_(this_=0,that_=0,oper="+") {
  if (this_ !== 0 && (this_ > 0 == false) && (this_ < 0 == false) || that_ !== 0 && (that_ > 0 == false ) && (that_ < 0 == false) ) {
    console.log("\nError >>> NaN parameter..\n");
    return 0;
  }
  if (typeof this_ != 'number' || typeof that_ != 'number') {
    return 0;
  }
 
  let answ = 0, a = this_, b = that_;
  const func = {
    "+": (a,b) => a + b,
    "-": (a,b) => a - b,
    "/": (a,b) => a / b,
    "*": (a,b) => a * b,
    "**": (a,b) => Math.pow(a,b),
    "_both": (a,b) => Math.floor(a/b),
    "_a": (a,b) => Math.floor(a),
    
  }
  
  answ = func[oper.trim()](a,b);
  return answ;
}
